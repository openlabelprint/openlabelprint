36  import java.net.URL;
37  
38  import org.dom4j.Document;
39  import org.dom4j.Element;
40  import org.dom4j.io.SAXReader;
41  import org.jaxen.dom4j.Dom4jXPath;
42  
43  class DOM4JPerformance {
44      
45      public static void main(String[] args) {
46          try {
47              URL u = new URL("http://www.ibiblio.org/xml/examples/shakespeare/much_ado.xml");
48              Document doc = new SAXReader().read(u);
49              Dom4jXPath xpath = new Dom4jXPath("PLAY/ACT/SCENE/SPEECH/SPEAKER");
50              
51              long start = System.currentTimeMillis();
52              
53              int count = 0;
54              for (int i = 0; i < 1000; i++) {
55                  Element speaker = (Element) xpath.selectSingleNode(doc);
56                  count += (speaker == null ? 0 : 1);
57              }
58              
59              long end = System.currentTimeMillis();
60              System.out.println((end - start));
61              System.out.println(count);
62              
63          } catch (Exception ex) {
64              ex.printStackTrace();
65          }
66      }
