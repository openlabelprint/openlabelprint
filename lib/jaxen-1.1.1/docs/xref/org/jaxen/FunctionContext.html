49  package org.jaxen;
50  
51  /*** Implemented by classes that know how to resolve XPath function names and
52   *  namespaces to implementations of these functions.
53   *
54   *  <p>
55   *  By using a custom <code>FunctionContext</code>, new or different
56   *  functions may be installed and available to XPath expression writers.
57   *  </p>
58   *
59   *  @see XPathFunctionContext
60   *
61   *  @author <a href="mailto:bob@werken.com">bob mcwhirter</a>
62   */
63  public interface FunctionContext
64  {
65      /*** An implementation should return a <code>Function</code> implementation object
66       *  based on the namespace URI and local name of the function-call
67       *  expression.
68       *
69       *  <p>
70       *  It must not use the prefix parameter to select an implementation,
71       *  because a prefix could be bound to any namespace; the prefix parameter
72       *  could be used in debugging output or other generated information.
73       *  The prefix may otherwise be completely ignored.
74       *  </p>
75       *
76       *  @param namespaceURI  the namespace URI to which the prefix parameter
77       *                       is bound in the XPath expression. If the function
78       *                       call expression had no prefix, the namespace URI
79       *                       is <code>null</code>.
80       *  @param prefix        the prefix that was used in the function call
81       *                       expression
82       *  @param localName     the local name of the function-call expression.
83       *                       If there is no prefix, then this is the whole
84       *                       name of the function.
85       *
86       *  @return  a Function implementation object.
87       *  @throws UnresolvableException  when the function cannot be resolved
88       */
89      Function getFunction( String namespaceURI,
90                            String prefix,
91                            String localName ) throws UnresolvableException;
92  }
