49  package org.jaxen.function;
50  
51  import java.util.List;
52  
53  import org.jaxen.Context;
54  import org.jaxen.Function;
55  import org.jaxen.FunctionCallException;
56  import org.jaxen.Navigator;
57  
58  /***
59   * <p><b>4.4</b> <code><i>number</i> round(<i>number</i>)</code> </p>
60   * 
61   * 
62   * <blockquote src="http://www.w3.org/TR/xpath">
63   * The <b>round</b> function returns the number that is closest to the argument and that is an integer. 
64   * If there are two such numbers, then the one that is closest to positive infinity is returned. 
65   * If the argument is NaN, then NaN is returned. If the argument is positive infinity, then positive infinity is returned. 
66   * If the argument is negative infinity, then negative infinity is returned. 
67   * If the argument is positive zero, then positive zero is returned. 
68   * If the argument is negative zero, then negative zero is returned. 
69   * If the argument is less than zero, but greater than or equal to -0.5, then negative zero is returned.
70   * </blockquote>
71   * 
72   * @author bob mcwhirter (bob @ werken.com)
73   * 
74   * @see <a href="http://www.w3.org/TR/xpath#function-round" target="_top">Section 4.4 of the XPath Specification</a>
75   */
76  public class RoundFunction implements Function
77  {
78   
79      /***
80       * Create a new <code>RoundFunction</code> object.
81       */
82      public RoundFunction() {}
83      
84      
85      /*** Returns the nearest integer to the number.
86       *
87       * @param context the context at the point in the
88       *         expression when the function is called
89       * @param args a list with exactly one item which will be converted to a 
90       *     <code>Double</code> as if by the XPath <code>number()</code> function
91       * 
92       * @return a <code>Double</code> containing the integer nearest to
93       *     <code>args.get(0)</code>
94       * 
95       * @throws FunctionCallException if <code>args</code> has more or less than one item
96       */
97      public Object call(Context context,
98                         List args) throws FunctionCallException
99      {
100         if (args.size() == 1)
101         {
102             return evaluate( args.get(0),
103                              context.getNavigator() );
104         }
105 
106         throw new FunctionCallException( "round() requires one argument." );
107     }
108 
109     /*** 
110      * Returns the integer nearest to the argument.
111      * If necessary, the argument is first converted to a <code>Double</code>
112      * as if by the XPath <code>number()</code> function.
113      * 
114      * @param obj the object to be rounded
115      * @param nav ignored
116      * 
117      * @return the integer nearest to <code>obj</code>
118      */
119     public static Double evaluate(Object obj,
120                                   Navigator nav)
121     {
122         Double d  = NumberFunction.evaluate( obj,
123                                             nav );
124 
125         if (d.isNaN() || d.isInfinite())
126         {
127             return d;
128         }
129 
130         double value = d.doubleValue();
131         return new Double( Math.round( value ) );
132     }
133 }
