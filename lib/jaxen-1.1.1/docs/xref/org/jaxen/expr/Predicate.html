50  package org.jaxen.expr;
51  
52  import java.io.Serializable;
53  import org.jaxen.Context;
54  import org.jaxen.JaxenException;
55  
56  
57  /***
58   * Represents an XPath predicate such as <code>[position() = last()]</code>.
59   * This is production 8 and 9 in the 
60   * <a href="http://www.w3.org/TR/xpath#NT-Predicate">XPath 1.0 specification</a>:
61   * 
62   * <pre> [8] Predicate     ::= '[' PredicateExpr ']'   
63   * [9] PredicateExpr ::= Expr</pre>
64   * 
65   */
66  public interface Predicate extends Serializable
67  {
68      /***
69       * Returns the expression in this predicate.
70       * 
71       * @return the expression between the brackets
72       */
73      Expr getExpr();
74      
75      /***
76       * Change the expression used by this predicate.
77       * 
78       * @param expr the new expression
79       */
80      void setExpr(Expr expr);
81  
82      /***
83       * Simplify the expression in this predicate.
84       * 
85       * @see Expr#simplify()
86       */
87      void simplify();
88  
89      /***
90       * Returns the string form of the predicate, 
91       * including the square brackets.
92       * 
93       * @return the bracketed form of this predicate
94       */
95      String getText();
96  
97      /***
98       * Evaluates this predicate's expression and returns the result.
99       * The result will be a <code>java.lang.Double</code> for expressions that 
100      * return a number, a <code>java.lang.String</code> for expressions that 
101      * return a string, a <code>java.lang.Boolean</code> for expressions that 
102      * return a boolean, and a <code>java.util.List</code> for expressions that
103      * return a node-set. In the latter case, the elements of the list are 
104      * the actual objects from the source document model. Copies are not made.
105      *
106      * @param context the context in which the expression is evaluated
107      * @return an object representing the result of the evaluation
108      * @throws JaxenException
109      * @see Expr#evaluate(Context)
110      */
111     Object evaluate(Context context) throws JaxenException;
112 
113 }
